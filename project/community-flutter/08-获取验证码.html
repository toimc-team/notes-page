class NumberCount {
  final int sum;
  final int duration;
  int _count = 1;
  Timer? _timer;

  StreamController<int> _stream = StreamController<int>();
  Stream<int> get stream => _stream.stream;
  // åˆ¤æ–­æ˜¯å¦å¯èƒ½è¿›è¡Œå†æ¬¡è¿›è¡Œè®¡æ•°
